\subsection{Problem~2}%
\label{problem:2}

Solve the following system of nonlinear equations with the selected nonlinear
least squares methods.

\begin{align*}
  F_1(\matr{x}) &= x_1^3 + 3x_1^2 + 3x_1 - x_2 - 3 = 0, \\
  F_2(\matr{x}) &= x_1^2 + 2x_1 - x_2 = 0.
\end{align*}

Draw the contour lines of the cost function in the nonlinear least squares problem.
Check the optimality points.

\subsubsection*{Mathematics}

Both methods minimise $f(\matr{x}) = \tfrac{1}{2}\|F(\matr{x})\|_2^2$ by iterating
a linearised subproblem \citep{NoceWrig99}.

\textbf{Gauss--Newton (GN)} approximates the Hessian by $J^T J$, discarding the
second-order correction $\sum_i F_i \nabla^2 F_i$:
\begin{equation*}
  \matr{x}_{k+1} = \matr{x}_k
    - \bigl(J(\matr{x}_k)^T J(\matr{x}_k)\bigr)^{-1} J(\matr{x}_k)^T F(\matr{x}_k).
\end{equation*}
The method converges quadratically near a zero of $F$ where $J$ has full column
rank, but may stagnate at a spurious local minimum where $J^T J$ is singular.

\textbf{Levenberg--Marquardt (LM)} regularises the normal equations with a damping
parameter $\lambda_k > 0$ \citep{NoceWrig99}:
\begin{equation*}
  \bigl(J(\matr{x}_k)^T J(\matr{x}_k) + \lambda_k I\bigr)\matr{p}_k
      = -J(\matr{x}_k)^T F(\matr{x}_k),
  \qquad
  \matr{x}_{k+1} = \matr{x}_k + \matr{p}_k.
\end{equation*}
When $\lambda_k \to 0$ the step recovers the GN step; when $\lambda_k \to \infty$
it approaches a scaled steepest-descent step.  The parameter is decreased after a
successful step (reduction in $f$) and increased otherwise, giving the method
global convergence properties that GN lacks.

\subsubsection*{Solution}

The Jacobian of the system is identical in form to that of
Problem~\ref{problem:1}:
\begin{equation*}
  J(\matr{x}) =
  \begin{bmatrix}
    3(x_1+1)^2 & -1 \\
    2(x_1+1)   & -1
  \end{bmatrix}.
\end{equation*}
The stationarity condition $\nabla f = J^T F = \matr{0}$ therefore produces the
same three cases as before: $x_1 = -1$,\ $F(\matr{x}) = \matr{0}$, and
$x_1 = -\tfrac{1}{3}$.  The constraint $F_2 = -F_1$ (from $-F_1 - F_2 = 0$)
gives
\begin{equation*}
  x_2 = \frac{x_1^3 + 4x_1^2 + 5x_1 - 3}{2}.
\end{equation*}

\medskip
\noindent\textbf{Case $x_1 = -1$:}
\begin{equation*}
  x_2 = \frac{-1+4-5-3}{2} = -\tfrac{5}{2},
\end{equation*}
giving $\matr{x}^{(1)} = (-1,\,-\tfrac{5}{2})^T$ with
$F_1 = -\tfrac{3}{2}$, $F_2 = \tfrac{3}{2}$.
Since $x_1^{(1)}+1 = 0$, the first column of $J$ vanishes, so $J^T J =
\begin{bmatrix}0&0\\0&2\end{bmatrix}$.
The second-order correction from $F_2\nabla^2 F_2$ (note $\nabla^2 F_1 = \matr{0}$
at $x_1 = -1$) yields
\begin{equation*}
  \nabla^2 f(\matr{x}^{(1)}) =
  \begin{bmatrix}0&0\\0&2\end{bmatrix}
  + \tfrac{3}{2}\begin{bmatrix}2&0\\0&0\end{bmatrix}
  = \begin{bmatrix}3&0\\0&2\end{bmatrix},
  \qquad M_1 = 3>0,\quad \det = 6>0.
\end{equation*}
The Hessian is positive definite but $f(\matr{x}^{(1)}) = \tfrac{9}{4} > 0$:
\textbf{spurious local minimum}.

\medskip
\noindent\textbf{Case $F_1 = 0$ (hence $F_2 = 0$):}
Subtracting the two equations gives $x_1^3 + 2x_1^2 + x_1 - 3 = 0$.
Substituting $u = x_1 + 1$ yields $u^3 - u^2 - 3 = 0$, whose unique real root
is $u^* \approx 1.8637$, so
\begin{equation*}
  \matr{x}^* \approx (0.8637,\;2.4734)^T.
\end{equation*}
Since $F(\matr{x}^*) = \matr{0}$ the Hessian reduces to $J(\matr{x}^*)^T J(\matr{x}^*)$,
which is positive definite, and $f = 0$: \textbf{global minimum}.

\medskip
\noindent\textbf{Case $x_1 = -\tfrac{1}{3}$:}
\begin{equation*}
  x_2 = \frac{-\tfrac{1}{27}+\tfrac{4}{9}-\tfrac{5}{3}-3}{2}
       = -\frac{115}{54},
\end{equation*}
giving $\matr{x}^{(2)} = \bigl(-\tfrac{1}{3},\,-\tfrac{115}{54}\bigr)^T$
with $F_1 = -\tfrac{85}{54}$, $F_2 = \tfrac{85}{54}$.
The rows of $J$ are equal at $x_1 = -\tfrac{1}{3}$, so $J^TJ$ is rank-deficient.
Including the second-order correction:
\begin{equation*}
  \nabla^2 f(\matr{x}^{(2)}) =
  \begin{bmatrix}\tfrac{11}{27} & -\tfrac{8}{3} \\[3pt] -\tfrac{8}{3} & 2\end{bmatrix},
  \qquad M_1 = \tfrac{11}{27}>0,\quad \det = -\tfrac{170}{27}<0:
\end{equation*}
\textbf{saddle point}.
