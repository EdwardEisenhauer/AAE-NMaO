\subsection{Problem~1}%
\label{problem:1}

Show that the following system of nonlinear equations:

\begin{align*}
  F_1(\matr{x}) &= x_1^3 + 3x_1^2 + 3x_1 - x_2 = 0, \\
  F_2(\matr{x}) &= x_1^2 + 2x_1 - x_2 + 1 = 0.
\end{align*}

has the global minimum at $\matr{x}^* = (0.46557 \quad 2.1479)^T$, a local minimum at
$\matr{x}^{(1)} = (-1 \quad {-0.5})^T$, and a saddle point at
$\matr{x}^{(2)} = \bigl(-\tfrac{1}{3} \quad -\tfrac{7}{54}\bigr)^T$.

\subsubsection*{Mathematics}

Solving a system of \emph{nonlinear equations} $F(\matr{x}) = \matr{0}$
means finding $\matr{x}^*$ such that every component $F_i(\matr{x}^*) = 0$.
The major algorithms proceed by minimising the \emph{objective function} \citep{Zdunek}:
\begin{equation*}
  f(\matr{x}) = \tfrac{1}{2}\bigl\|F(\matr{x})\bigr\|_2^2
              = \tfrac{1}{2}\sum_{i=1}^{N} F_i^2(\matr{x}).
\end{equation*}
Any zero of $F$ is a global minimum of $f$ with $f(\matr{x}^*) = 0$.
However, $f$ may also admit \emph{spurious local minima} with $f > 0$,
where $F(\matr{x}) \neq \matr{0}$ \citep{NoceWrig99}.

The \textbf{Jacobian} of $F$ collects all first partial derivatives:
\begin{equation*}
  J(\matr{x}) = \left[\frac{\partial F_i}{\partial x_j}\right] \in \mathbb{R}^{N \times N}.
\end{equation*}
The gradient and Hessian of the objective function are expressed in terms of $J$ as:
\begin{align*}
  \nabla f(\matr{x})   &= J(\matr{x})^T F(\matr{x}), \\
  \nabla^2 f(\matr{x}) &= J(\matr{x})^T J(\matr{x})
                          + \sum_{i=1}^{N} F_i(\matr{x})\,\nabla^2 F_i(\matr{x}).
\end{align*}
Note that at a zero of $F$, the second term vanishes and the Hessian
reduces to $J^T J$.

Stationary points satisfy $\nabla f = \matr{0}$.
Their nature is determined by \textbf{Sylvester's criterion}: the Hessian
$\nabla^2 f$ is positive definite if and only if all leading principal minors
$M_k > 0$.
A stationary point is:
\begin{itemize}
  \item a \textbf{global minimum} if $\det(\nabla^2 f) > 0$, $M_1 > 0$, and $f = 0$
        (a true solution of the system);
  \item a \textbf{spurious local minimum} if $\det(\nabla^2 f) > 0$, $M_1 > 0$,
        and $f > 0$;
  \item a \textbf{saddle point} if $\det(\nabla^2 f) < 0$.
\end{itemize}

\subsubsection*{Solution}

The Jacobian of the given system is:
\begin{equation*}
  J(\matr{x}) =
  \begin{bmatrix}
    3x_1^2 + 6x_1 + 3 & -1 \\
    2x_1 + 2           & -1
  \end{bmatrix}
  =
  \begin{bmatrix}
    3(x_1+1)^2 & -1 \\
    2(x_1+1)   & -1
  \end{bmatrix}.
\end{equation*}

The stationarity condition $\nabla f = J^T F = \matr{0}$ expands to:
\begin{align*}
  3(x_1+1)^2\,F_1 + 2(x_1+1)\,F_2 &= 0, \\
  -F_1 - F_2 &= 0.
\end{align*}
The second condition gives $F_2 = -F_1$; substituting into the first:
\begin{equation*}
  (x_1+1)\,F_1\,(3x_1+1) = 0.
\end{equation*}
This product yields three independent cases.

\medskip
\noindent\textbf{Case $x_1 = -1$:}
Using $F_2 = -F_1$ to solve for $x_2$:
\begin{equation*}
  x_2 = \frac{x_1^3 + 4x_1^2 + 5x_1 + 1}{2}\bigg|_{x_1=-1} = -\tfrac{1}{2},
\end{equation*}
giving the stationary point $\matr{x}^{(1)} = (-1,\,-\tfrac{1}{2})^T$.
Since $x_1^{(1)} + 1 = 0$, the Jacobian has zero left columns, so
$J^T J = \begin{bmatrix} 0 & 0 \\ 0 & 2 \end{bmatrix}$.
With $F_1(\matr{x}^{(1)}) = -\tfrac{1}{2}$ and $F_2(\matr{x}^{(1)}) = \tfrac{1}{2}$:
\begin{equation*}
  \nabla^2 f(\matr{x}^{(1)}) =
  \begin{bmatrix} 0 & 0 \\ 0 & 2 \end{bmatrix}
  + F_2 \nabla^2 F_2 =
  \begin{bmatrix} 1 & 0 \\ 0 & 2 \end{bmatrix},\qquad
  M_1 = 1 > 0,\quad \det = 2 > 0.
\end{equation*}
The Hessian is positive definite, but $f(\matr{x}^{(1)}) = \tfrac{1}{4} > 0$:
\textbf{spurious local minimum}.

\medskip
\noindent\textbf{Case $F_1 = 0$ (hence $F_2 = 0$):}
Subtracting the two equations eliminates $x_2$:
\begin{equation*}
  x_1^3 + 2x_1^2 + x_1 - 1 = 0.
\end{equation*}
This cubic has a unique real root $x_1^* \approx 0.46557$, with
$x_2^* = (x_1^*+1)^2 \approx 2.1479$, giving $\matr{x}^* \approx (0.46557,\,2.1479)^T$.
Since $F(\matr{x}^*) = \matr{0}$, the Hessian reduces to $J(\matr{x}^*)^T J(\matr{x}^*)$:
\begin{equation*}
  \nabla^2 f(\matr{x}^*) \approx
  \begin{bmatrix} 50.113 & -9.375 \\ -9.375 & 2 \end{bmatrix},\qquad
  M_1 = 50.113 > 0,\quad \det = 12.34 > 0.
\end{equation*}
The Hessian is positive definite and $f(\matr{x}^*) = 0$:
\textbf{global minimum} --- a true solution of the system.

\medskip
\noindent\textbf{Case $x_1 = -\tfrac{1}{3}$:}
From the constraint $F_2 = -F_1$:
\begin{equation*}
  x_2 = \frac{x_1^3 + 4x_1^2 + 5x_1 + 1}{2}\bigg|_{x_1=-1/3} = -\tfrac{7}{54},
\end{equation*}
giving $\matr{x}^{(2)} = (-\tfrac{1}{3},\,-\tfrac{7}{54})^T$.
At this point $F_1 = -\tfrac{31}{54}$, $F_2 = \tfrac{31}{54}$:
\begin{equation*}
  \nabla^2 f(\matr{x}^{(2)}) =
  \begin{bmatrix} \tfrac{65}{27} & -\tfrac{8}{3} \\[3pt] -\tfrac{8}{3} & 2 \end{bmatrix},\qquad
  M_1 = \tfrac{65}{27} > 0,\quad \det = -\tfrac{62}{27} < 0.
\end{equation*}
The Hessian is indefinite: \textbf{saddle point}.
